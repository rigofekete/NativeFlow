# Minimum required version of CMake
cmake_minimum_required(VERSION 3.10)

# Add vcpkg toolchain file
set(CMAKE_TOOLCHAIN_FILE "C:/Dev/vcpkg/scripts/buildsystems/vcpkg.cmake")

# Project name
project(NativeFlow)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)

add_definitions(-DUNINativeFlow -D_UNICODE)

# Set up static linking for Boost
set(Boost_USE_STATIC_LIBS ON)
find_package(Boost REQUIRED COMPONENTS serialization)

# Collect all source files in the src directory
# Also adding the resources.rc file here (necessary to set up a custom icon *.ico file)
file(GLOB SRC_FILES ${CMAKE_SOURCE_DIR}/src/*.cpp ${CMAKE_SOURCE_DIR}/src/resources.rc)

# Add the collected source files to the executable
add_executable(NativeFlow ${SRC_FILES})


set_target_properties(NativeFlow PROPERTIES
    WIN32_EXECUTABLE TRUE    # This tells CMake it's a Windows GUI app
)

# Specify where to find header files
target_include_directories(NativeFlow 
    PRIVATE 
        ${CMAKE_SOURCE_DIR}/headers
        # include the path for the Boost headers
        ${Boost_INCLUDE_DIRS}
) 

# Adding this to temporarily disable the warning I am getting for these type converions 
# I will eventually come back to this to fix them and then deactivate this  
if(MSVC)
    # Disable specific MSVC warnings for the target
    target_compile_options(NativeFlow PRIVATE
        /wd4244  # Disable warning C4244: conversion from 'type1' to 'type2', possible loss of data
        /wd4267  # Disable warning C4267: conversion from 'size_t' to 'type', possible loss of data
    )

    target_compile_definitions(NativeFlow PRIVATE
        NOMINMAX  # Disable Windows min/max macros
    )
endif()

# Link the GDI32 library and Boost
target_link_libraries(NativeFlow
    PRIVATE
        gdi32
        Boost::serialization
        comctl32
        dwmapi
)
